\subsection{Problem 2.14.2}
Kursusgang 4.\\
Given the vectors $v=[2,-1,5]$ and $u=[-1,1,1]$ find the vectors $v+u$, $v-u$, $2v-u$, and $v+u+u$
\begin{itemize}
    \item \begin{equation}
        v+u=[1,0,6]
    \end{equation}
    \item \begin{equation}
        v-u=[3,-2,4]
    \end{equation}
    \item \begin{equation}
        2v-u=[5,-3, 9]
    \end{equation}
    \item \begin{equation}
        v+u+u=[0,1,7]
    \end{equation}
\end{itemize}
